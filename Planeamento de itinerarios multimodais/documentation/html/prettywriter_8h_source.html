150  void PrettyPrefix(Type type) {
151  (void)type;
152  if (Base::level_stack_.GetSize() != 0) { // this value is not at root
153  typename Base::Level* level = Base::level_stack_.template Top<typename Base::Level>();
154 
155  if (level->inArray) {
156  if (level->valueCount > 0) {
157  Base::os_->Put(','); // add comma if it is not the first element in array
158  Base::os_->Put('\n');
159  }
160  else
161  Base::os_->Put('\n');
162  WriteIndent();
163  }
164  else { // in object
165  if (level->valueCount > 0) {
166  if (level->valueCount % 2 == 0) {
167  Base::os_->Put(',');
168  Base::os_->Put('\n');
169  }
170  else {
171  Base::os_->Put(':');
172  Base::os_->Put(' ');
173  }
174  }
175  else
176  Base::os_->Put('\n');
177 
178  if (level->valueCount % 2 == 0)
179  WriteIndent();
180  }
181  if (!level->inArray && level->valueCount % 2 == 0)
182  RAPIDJSON_ASSERT(type == kStringType); // if it's in object, then even number should be a name
183  level->valueCount++;
184  }
185  else {
186  RAPIDJSON_ASSERT(!Base::hasRoot_); // Should only has one and only one root.
187  Base::hasRoot_ = true;
188  }
189  }
